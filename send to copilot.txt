<script>
/* STATE */
let balance = 0;
let basePerClick = 1;
let incomeMultiplier = 1; // affects clicks + utilities
let cursorBots = [];
let lastUpdateTime = Date.now();
let gardenStartTime = Date.now();
let buffTimeoutId = null;
let rainbowCoinActive = false;

const SAVE_KEY = "crypto_clicker_pixel_emoji_rainbow_v2";

/* PRODUCERS */
let producers = [
  { name: "Cursor Bot", baseCost: 15, amount: 0, cps: 0.1, emoji: "🖱️" },
  { name: "Hobby Miner", baseCost: 100, amount: 0, cps: 1, emoji: "⛏️" },
  { name: "Mining Farm", baseCost: 1100, amount: 0, cps: 8, emoji: "🏭" },
  { name: "GPU Factory", baseCost: 12000, amount: 0, cps: 47, emoji: "💻" },
  { name: "Crypto Mine", baseCost: 130000, amount: 0, cps: 260, emoji: "🪨" },
  { name: "Server Shipment", baseCost: 1400000, amount: 0, cps: 1400, emoji: "📦" },
  { name: "Hash Lab", baseCost: 20000000, amount: 0, cps: 7800, emoji: "⚗️" },
  { name: "Quantum Gateway", baseCost: 330000000, amount: 0, cps: 44000, emoji: "🌀" },
  { name: "Time Miner", baseCost: 5100000000, amount: 0, cps: 260000, emoji: "⏳" }
];

/* COST */
function getCost(base, amount) {
  return Math.floor(base * Math.pow(1.15, amount));
}

/* UI UPDATE */
function updateUI() {
  document.getElementById("balance").textContent = balance.toFixed(1) + " Crypto";

  const shop = document.getElementById("shop");
  shop.innerHTML = "";

  producers.forEach((p, i) => {
    const cost = getCost(p.baseCost, p.amount);

    const item = document.createElement("div");
    item.className = "item" + (balance < cost ? " disabled" : "");
    item.onclick = () => buyProducer(i);

    const icon = document.createElement("div");
    icon.className = "item-icon";
    icon.textContent = p.emoji;

    const text = document.createElement("div");
    text.innerHTML = `
      <div style="font-weight:bold;color:#aaffc4">${p.name} (${p.amount})</div>
      <div style="font-size:12px;color:#7aff9d">Generates ${p.cps} crypto/sec</div>
      <div style="color:#00ff66;font-weight:bold">Cost: ${cost.toLocaleString()}</div>
    `;

    item.appendChild(icon);
    item.appendChild(text);
    shop.appendChild(item);
  });

  renderGarden();
  updateBuffStatus();
}

/* BUFF STATUS TEXT */
function updateBuffStatus() {
  const el = document.getElementById("buffStatus");
  if (incomeMultiplier > 1) {
    el.textContent = "Rainbow Boost: 10× clicks & utilities for 30s!";
  } else {
    el.textContent = "";
  }
}

/* BUY */
function buyProducer(i) {
  const p = producers[i];
  const cost = getCost(p.baseCost, p.amount);
  if (balance < cost) return;

  balance -= cost;
  p.amount++;

  if (i === 0) spawnCursorBot();

  updateUI();
  saveGame();
}

/* CLICK */
document.getElementById("bigCoin").onclick = () => {
  balance += basePerClick * incomeMultiplier;
  updateUI();
  saveGame();
};

/* CURSOR BOTS */
const cursorSVG = `
<svg viewBox="0 0 24 24">
  <path d="M3 2 L20 12 L13 13 L15 22 L10 22 L8 13 L3 2 Z"
        fill="black" stroke="#00ff66" stroke-width="2"/>
</svg>`;

function spawnCursorBot() {
  const bot = document.createElement("div");
  bot.className = "cursorBot";
  bot.innerHTML = cursorSVG;
  document.body.appendChild(bot);

  cursorBots.push({
    el: bot,
    angle: Math.random() * Math.PI * 2,
    radius: 170 + cursorBots.length * 12,
    speed: 0.01 + Math.random() * 0.01
  });
}

function animateBots() {
  const coin = document.getElementById("bigCoin");
  const rect = coin.getBoundingClientRect();
  const cx = rect.left + rect.width / 2;
  const cy = rect.top + rect.height / 2;

  cursorBots.forEach(bot => {
    bot.angle += bot.speed;
    const x = cx + Math.cos(bot.angle) * bot.radius - 16;
    const y = cy + Math.sin(bot.angle) * bot.radius - 16;
    bot.el.style.transform = `translate(${x}px, ${y}px)`;
  });

  requestAnimationFrame(animateBots);
}
animateBots();

/* PASSIVE INCOME (AFFECTED BY RAINBOW BUFF) */
setInterval(() => {
  const now = Date.now();
  const delta = (now - lastUpdateTime) / 1000;
  lastUpdateTime = now;

  let total = 0;
  producers.forEach(p => total += p.amount * p.cps);
  balance += total * delta * incomeMultiplier;

  updateUI();
  saveGame();
}, 1000);

/* UTILITIES PANEL */
document.getElementById("utilitiesBtn").onclick = () =>
  document.getElementById("utilitiesPanel").classList.add("open");

document.getElementById("closeUtilities").onclick = () =>
  document.getElementById("utilitiesPanel").classList.remove("open");

/* EVOLUTION */
function getEvolutionLevel() {
  const minutes = (Date.now() - gardenStartTime) / 60000;
  return Math.min(3, Math.floor(minutes / 30));
}

/* GARDEN RENDER */
function renderGarden() {
  const grid = document.getElementById("gardenGrid");
  grid.innerHTML = "";

  const level = getEvolutionLevel();

  producers.forEach(p => {
    if (p.amount <= 0) return;

    const row = document.createElement("div");
    row.className = "garden-row";

    for (let i = 0; i < p.amount; i++) {
      const tile = document.createElement("div");
      tile.className = "garden-tile level-" + level;

      const emoji = document.createElement("div");
      emoji.className = "tile-emoji";
      emoji.textContent = p.emoji;

      tile.appendChild(emoji);
      row.appendChild(tile);
    }

    grid.appendChild(row);
  });
}

/* RAINBOW FALLING COIN LOGIC */
function spawnRainbowCoin() {
  if (rainbowCoinActive) return;
  rainbowCoinActive = true;

  const coin = document.createElement("div");
  coin.className = "rainbow-coin";
  coin.textContent = "₿";

  const screenWidth = window.innerWidth;
  const x = Math.random() * (screenWidth - 60) + 10;
  coin.style.left = x + "px";

  document.body.appendChild(coin);

  const fallDuration = 4000; // 4 seconds
  const startTime = performance.now();

  function fallStep(now) {
    const t = (now - startTime) / fallDuration;
    if (!document.body.contains(coin)) return;

    const y = t * (window.innerHeight + 80) - 60;
    coin.style.top = y + "px";

    if (t >= 1) {
      // fell off screen
      document.body.removeChild(coin);
      rainbowCoinActive = false;
      return;
    }
    requestAnimationFrame(fallStep);
  }
  requestAnimationFrame(fallStep);

  coin.addEventListener("click", () => {
    if (!document.body.contains(coin)) return;
    document.body.removeChild(coin);
    rainbowCoinActive = false;
    triggerRainbowBuff();
  });
}

function triggerRainbowBuff() {
  incomeMultiplier = 10; // 10x clicks + utilities
  updateBuffStatus();

  if (buffTimeoutId) clearTimeout(buffTimeoutId);
  buffTimeoutId = setTimeout(() => {
    incomeMultiplier = 1;
    buffTimeoutId = null;
    updateBuffStatus();
  }, 30000); // 30 seconds
}

/* RANDOM SPAWN CHECK: 1 IN 100 CHANCE EVERY SECOND */
setInterval(() => {
  const roll = Math.floor(Math.random() * 100); // 0–99
  if (roll === 0) {
    spawnRainbowCoin();
  }
}, 1000);

/* SAVE / LOAD */
function saveGame() {
  localStorage.setItem(SAVE_KEY, JSON.stringify({
    balance,
    producers: producers.map(p => p.amount),
    lastUpdateTime,
    gardenStartTime
  }));
}

function loadGame() {
  const raw = localStorage.getItem(SAVE_KEY);
  if (!raw) {
    lastUpdateTime = Date.now();
    gardenStartTime = Date.now();
    return;
  }
  const data = JSON.parse(raw);

  balance = data.balance || 0;
  if (Array.isArray(data.producers)) {
    data.producers.forEach((amt, i) => {
      if (producers[i]) producers[i].amount = amt || 0;
    });
  }

  lastUpdateTime = data.lastUpdateTime || Date.now();
  gardenStartTime = data.gardenStartTime || Date.now();

  const delta = (Date.now() - lastUpdateTime) / 1000;
  let total = 0;
  producers.forEach(p => total += p.amount * p.cps);
  balance += total * delta; // offline income (no buff while closed)

  for (let i = 0; i < producers[0].amount; i++) spawnCursorBot();
}

loadGame();
updateUI();
</script>
